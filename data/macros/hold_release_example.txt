# Example macro demonstrating HOLD and RELEASE commands
# This macro shows how to hold buttons while performing other actions

# Hold ZR button (often used for running/shooting)
HOLD zl

# While holding ZR, move the character around
STICK l 0.5 0.8
SLEEP 2
STICK l -0.5 0.8  
SLEEP 2
STICK l 0.0 0.0

# Release ZR button
RELEASE zr
